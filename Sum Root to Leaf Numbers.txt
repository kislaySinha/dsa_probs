/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void rec(TreeNode* root, int sum, int& finalsum){
        if(root) sum= sum*10 + root->val;
        if(!root) return;
        if(root->right==NULL && root->left==NULL){
            finalsum+=sum;
            return;
        }
        
        rec(root->left, sum, finalsum);
        rec(root->right, sum, finalsum);
    }
    int sumNumbers(TreeNode* root) {
        int finalsum=0;
        rec(root,0,finalsum);
        return finalsum;
    }
};
